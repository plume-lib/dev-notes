  preplace "  description ([^=])" "  description = \1" build.gradle
  if [ -f gradle/mavencentral.gradle ] ; then
    preplace "^group ([^=])" "group = \1" gradle/mavencentral.gradle
    preplace "^version ([^=])" "version = \1" gradle/mavencentral.gradle
    git commit -m "Fix gradle warnings"  build.gradle gradle/mavencentral.gradle
  else
    git commit -m "Fix gradle warnings"  build.gradle
  fi
  git push
  xdg-open https://github.com/mernst/$project/pull/new/$BRANCHNAME
done

# To add a new version of Java:
# Note: You might need to use the "-ea" version for a while even after the GA release.
# Before upgrading Java:
# * update to a Gradle that supports the given Java version:
#   https://docs.gradle.org/current/userguide/compatibility.html .
VER_PREV=26
# VER_NEXT=27
VER_NEXT=27-ea-2
BRANCHNAME=java-${VER_NEXT}
for project in bcel-util bibtex-clean hashmap-util html-pretty-print icalavailable javac-parse javadoc-lookup lookup merging multi-version-control options plume-util reflection-util require-javadoc ; do
  cd ~/java/plume-lib/$project-fork-mernst
  gnb $BRANCHNAME
  cd ~/java/plume-lib/$project-fork-mernst-branch-$BRANCHNAME
  pwd
  sed -i "s/'${VER_PREV}'/'${VER_PREV}', '${VER_NEXT}'/g" .github/workflows/gradle.yml
  git add .github/workflows/gradle.yml
  if [ -f .github/workflows/gradle-native-compile.yml ] ; then
    sed -i "s/'${VER_PREV}'/'${VER_PREV}', '${VER_NEXT}'/g" .github/workflows/gradle-native-compile.yml
    git add .github/workflows/gradle-native-compile.yml
  fi
  git commit -m "Test under JDK ${VER_NEXT}"
  git push
  xdg-open https://github.com/mernst/$project/pull/new/$BRANCHNAME
done

# To upgrade from an early release version of Java:
# This might replace a previous non-ea version, or not; see multiple sed commands.
VER=26
BRANCHNAME=java-${VER}-released
for project in bcel-util bibtex-clean hashmap-util html-pretty-print icalavailable javac-parse javadoc-lookup lookup merging multi-version-control options plume-util reflection-util require-javadoc ; do
  cd ~/java/plume-lib/$project-fork-mernst
  gnb $BRANCHNAME
  cd ~/java/plume-lib/$project-fork-mernst-branch-$BRANCHNAME
  pwd
  if [ -f .github/workflows/gradle-native-compile.yml ] ; then
    gradle_files=".github/workflows/gradle.yml .github/workflows/gradle-native-compile.yml"
  else
    gradle_files=".github/workflows/gradle.yml"
  fi
  # Which of these two `sed` commands is appropriate depends on the LTS version of Java.
  sed -i "s/'24', '25-ea'/'25'/g" ${gradle_files}
  sed -i "s/'25', '26-ea'/'25', '26'/g" ${gradle_files}
  git add ${gradle_files}
  git commit -m "Test under JDK ${VER}, not JDK ${VER}-ea"
  git push
  xdg-open https://github.com/mernst/$project/pull/new/$BRANCHNAME
done
echo "Click \"Create pull request\" in the browser."

# To use a gradle Version Catalog:
BRANCHNAME=gradle-version-catalog
for project in bcel-util bibtex-clean hashmap-util html-pretty-print icalavailable javac-parse javadoc-lookup lookup merging multi-version-control options plume-util reflection-util require-javadoc ; do
  cd ~/java/plume-lib/$project-fork-mernst
  gnb $BRANCHNAME
  cd ~/java/plume-lib/$project-fork-mernst-branch-$BRANCHNAME
  pwd
  preplace '(// Project name is read-only in build scripts)' 'plugins {    id("de.fayard.refreshVersions") version "0.60.6" } \1' settings.gradle
  ./gradlew refreshVersionsMigrate --mode=VersionCatalogOnly
  rm -f versions.properties
  preplace 'id\("(.*\..*)"\)' 'alias(libs.plugins.\1)' build.gradle
  preplace ':_' '' build.gradle
  cronic make
  cronic ./gradlew build && git checkout settings.gradle
done
cd ~/java/plume-lib
# Sort paragraphs in all files gradle/libs.versions.toml.  Here are the files:
find . -name libs.versions.toml | sort | grep $BRANCHNAME
